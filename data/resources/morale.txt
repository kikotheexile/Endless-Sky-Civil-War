# Events that affect morale:

# A crew member has died on a ship in the fleet
"morale affected" "death in fleet"
  change -2

# A crew member has died on the ship
# This happens in addition to "death in fleet"
"morale affected" "death on ship"
  change -3

# A crew member has been hired on a ship in the fleet
"morale affected" "hire in fleet"
  change 0

# A crew member has been hired on a ship in the fleet
# This happens in addition to "hire in fleet"
"morale affected" "hire on ship"
  change 0

# A crew member has been fired on a ship in the fleet
"morale affected" "fire in fleet"
  change -0.1

# A crew member has been fired on the ship; stacks with "fire in fleet"
# This happens in addition to "fire in fleet"
"morale affected" "fire on ship"
  change -1

# A crew member on a ship in the fleet has mutinied
"morale affected" "mutiny in fleet"
  change -0.1

# Profit has been shared with the crew on the ship
# Multiplied by the credits shared and divided by the ship's crew count
"morale affected" "profit shared"
  # 1 morale per 1000 credits, per crew member on the ship
  change 0.001

# The player has shared profits while the ship is parked
# Multiplied by the credits shared and divided by the ship's crew count
"morale affected" "profit shared while parked"
  # 1 morale per 100 credits, per crew member on the ship
  change 0.01

# The player has failed to pay crew salaries
"morale affected" "salary failure"
  change -10

# The player has paid crew salaries
"morale affected" "salary payment"
  change 0.1

# The player owes profit shares while the ship is parked
# Multiplied by the credits paid and divided by the ship's crew count
"morale affected" "profit owed while parked"
  change 0.01
